public static Vector union(Vector a, Vector b)
  { Vector res = new Vector();
    for (int i = 0; i < a.size(); i++)
    { Object aobj = a.elementAt(i);
      if (res.contains(aobj)) {}
      else { res.add(aobj); } 
    }
    for (int j = 0; j < b.size(); j++)
    { Object bobj = b.elementAt(j);
      if (res.contains(bobj)) {}
      else { res.add(bobj); } 
    }
    return res; 
  }
